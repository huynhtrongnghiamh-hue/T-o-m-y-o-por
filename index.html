def shorten_url(long_url):
    try:
        api_url = f"https://vuotnhanh.com/api?api={VUOTNHANH_API}&url={long_url}&format=text"
        response = requests.get(api_url, timeout=10)
        return response.text.strip() if response.status_code == 200 else long_url
    except:
        return long_url

def load_keys():
    global key_db
    if os.path.exists("keys.json"):
        with open("keys.json", "r", encoding="utf-8") as f:
            key_db = json.load(f)
    else:
        key_db = {"verified": {}, "keys": {}, "blacklist": [], "temp_keys": {}}
    save_keys()

def save_keys():
